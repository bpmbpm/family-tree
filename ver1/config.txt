# config.json — параметры конфигурации семейного дерева
# Все параметры можно изменять вручную в config.json.

# --- Папка с миниатюрами (фото персон) ---

picDir      — путь к папке с фото персон (по умолчанию: "pic")
              Можно указывать как относительный путь (например "pic", "assets/photos"),
              так и полный абсолютный путь (например "/home/user/photos").
              Значение по умолчанию: "pic" (папка pic рядом с index.html).

picDirType  — режим работы с папкой фото (по умолчанию: "relative")
              Два основных режима:
                "server"                — запуск с HTTP-сервера (GitHub Pages, локальный веб-сервер):
                                         локальное отображение использует picDir (относительный путь);
                                         кнопка «Показать в окне GraphvizOnline» подставляет
                                         picDirGraphvizOnline (полный URL), чтобы GraphvizOnline
                                         мог загрузить фото из интернета.
                                         Псевдоним: "relativeGraphvizOnline" (равнозначно "server").
                "desktop"               — запуск с локального диска через file://:
                                         использует picDirGlobal как полный абсолютный путь к фото.
                                         Псевдоним: "global" (равнозначно "desktop").
              Дополнительный режим:
                "relative"              — только относительный путь picDir; работает при HTTP-сервере,
                                         но фото не отображаются в GraphvizOnline.

picDirGlobal — полный абсолютный путь к папке с фото; используется в режиме "desktop" / "global"
              Пример для Windows: "C:/Users/YourName/Desktop/ver1/pic"
              Пример для Linux/Mac: "/home/user/family-tree/ver1/pic"
              Значение по умолчанию: "" (не задано).

picDirGraphvizOnline — URL к папке с фото для GraphvizOnline; используется в режиме "server" / "relativeGraphvizOnline"
              Пример: "https://bpmbpm.github.io/family-tree/ver1/pic"
              Значение по умолчанию: "" (не задано).

# --- Размеры узлов (в дюймах, при 72 DPI) ---

nodeWidth   — ширина узла в дюймах (по умолчанию: 1.1)
nodeHeight  — высота узла в дюймах (по умолчанию: 1.9)

# --- Шрифт ---

fontName    — название шрифта для подписей узлов (по умолчанию: "Arial")
              Graphviz поддерживает системные шрифты: Arial, Times New Roman, Courier New и др.
fontSize    — размер шрифта в пунктах (по умолчанию: 11)
lineSpacing — межстрочный интервал (по умолчанию: 1.0; значение 1.5 = полуторный интервал)
lineSpacingSurName2 — масштаб размера шрифта для узлов, у которых заполнено поле surName2 (по умолчанию: 0.8).
              Размер шрифта = fontSize × lineSpacingSurName2.
              Меньшее значение → меньший шрифт → более компактное отображение.
              Пример: fontSize=11, lineSpacingSurName2=0.8 → шрифт 8.8pt для таких узлов.

# --- Цвета узлов ---

maleColor   — цвет заливки узла для мужчин
              Значение: "lightsteelblue"  (светло-стальной синий)
              HEX-код:  #a3c4f3

femaleColor — цвет заливки узла для женщин
              Значение: "lightpink"        (светло-розовый)
              HEX-код:  #fbb9c0

unknownColor — цвет заливки узла, если пол не указан
              Значение: "lightgrey"        (светло-серый)
              HEX-код:  #d5d8dc

borderColor — цвет рамки (границы) узла
              Значение: "darkslategray"    (тёмно-серый сланец)
              HEX-код:  #2c3e50

edgeColor   — цвет стрелок и линий связей
              Значение: "slategray"        (серый сланец)
              HEX-код:  #5d6d7e

# --- Примечания ---
# Цвета можно задавать:
#   - именем (например "lightblue", "pink") — список имён: https://graphviz.org/doc/info/colors.html
#   - HEX-кодом (например "#a3c4f3")
# lineSpacingSurName2 применяется как множитель для fontSize узлов с заполненным surName2.
